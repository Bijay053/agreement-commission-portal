6. Bonus Calculation Logic (Backend)

Pseudo code:

def calculate_bonus(target, student_count, country=None):

    rule = get_bonus_rule(target)

    if rule.type == "tier_per_student":
        tier = find_matching_tier(student_count)

        if tier.calculation_type == "per_student":
            return student_count * tier.bonus_amount

    if rule.type == "flat":
        tier = find_matching_tier(student_count)
        return tier.bonus_amount

    if rule.type == "country":
        if country_students(country) >= rule.student_count:
            return rule.bonus_amount

    return 0
